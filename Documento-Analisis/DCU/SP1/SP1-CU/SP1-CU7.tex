%--------------------------- CU TRAYECTORIA PRINCIPAL -------------------------
\begin{UCtrayectoria}{Principal}
\UCpaso[\UCactor] Presiona el botón \IUbutton{Registrar autor/autores} de la interfaz de usuario \IUref{IU6}{Página Principal}

% \ UCpaso El sistema busca el formulario \ Formref {FM1} {Registrar autor/autores}.

\UCpaso Muestra la interfaz de usuario \IUref{IU6}{Registro del autor/autores}.
\UCpaso[\UCactor] Presiona el botón \IUbutton{Eliminar autor} de la interfaz de usuario \IUref{IU7}{Página Principal}
\UCpaso El sistema muestra el mensaje \MSGref {MSG1} {¿Seguro que deseas eliminar este autor?}.
\UCpaso[\UCactor] Termina la operación presionando el botón \IUbutton{Aceptar}. [Trayectoria A]
\UCpaso El sistema muestra el mensaje \MSGref{MSG2}{Registro eliminado exitosamente}.
\UCpaso[\UCactor] Cierra el MSG2 presionando el botón \IUbutton{Ok}.
\UCpaso Muestra la interfaz de usuario \IUref{IU6}{Página principal}.
\end{UCtrayectoria}

%------------------------ CU TRAYECTORIA ALTERNARTIVA A -------------------------

\begin{UCtrayectoriaA}{A}{El usuario desea cancelar la eliminación del autor de la bibliografía.}

\UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar}.
\UCpaso Continua en el paso 7 de la trayectoria principal del \UCref{CU7}.

\end{UCtrayectoriaA}
