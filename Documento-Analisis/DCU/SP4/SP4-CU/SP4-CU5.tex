\begin{UseCase}{SP4-CU5}{Consultar Programas Académicos}{El usuario Jefe de Innovación Educativa visualiza la información
de los Programas Académicos.}


		\UCitem{Versión}{\color{Gray}1.0}
		\UCitem{Autor}{\color{Gray}Plata García Josué Eliasaf}
		\UCitem{Supervisa}{\color{Gray}}
		\UCitem{Actor}{\hyperlink{Usuario}{Jefe de Innovación Educativa}}
		\UCitem{Propósito}{El Usuario Jefe de Innovación Educativa puede visualizar una lista  de  los Programas Académicos almacenados en el Sistema.}
		\UCitem{Entradas}{Las entradas para la consulta del Programa Académico:
          \begin{itemize}
          	\item Ninguna
          \end{itemize}
        }
		\UCitem{Origen}{Ninguno}
		\UCitem{Salidas}{
        	\begin{itemize}
        		\item MSG4. ¿Seguro que desea editar el Programa Académico?
                \item  MSG5. ¿Seguro que desea cancelar la consulta de Programas Académicos?
                \item  MSG6. ¿Seguro que desea registrar un Programa Académico?
                \item  MSG7. Por ahorita no se puede consultar.
        	\end{itemize}
        }
		\UCitem{Destino}{Pantalla.}
		\UCitem{Precondiciones}{El catálogo de tipo Programa Académico debe de estar cargado en el sistema.}
		\UCitem{Postcondiciones}{Ninguna}
		\UCitem{Errores}{E1. No se cargaron los catálogos de Programas Académicos.}
		\UCitem{Estado}{Revisión.}
		\UCitem{Observaciones}{}
\end{UseCase}

%--------------------------- CU TRAYECTORIA PRINCIPAL -------------------------
\begin{UCtrayectoria}{Principal}

    \UCpaso[\UCactor] Presiona el botón \IUbutton{Consultar Programa Académico} de la interfaz de usuario menú para Jefe de Innovación Educativa de Abi

    \UCpaso Muestra la interfaz de usuario \IUref{IU2.2-J}{Consultar Programa Académico}.
    \UCpaso[\UCactor] El sistema muestra la lista de Programas Académicos registrados. [Trayectoria A][Trayectoria B][Trayectoria C][Trayectoria D][Trayectoria E][Trayectoria F][Trayectoria G]
\end{UCtrayectoria}


%------------------------ CU TRAYECTORIA ALTERNARIVA A -------------------------


\begin{UCtrayectoriaA}{A}{ El actor presiona el botón cancelar.}
	\UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar}.
    \UCpaso Muestra el mensaje \MSGref{MSG5}{¿Seguro que desea cancelar la consulta de Programas Académicos?}.
    \UCpaso[\UCactor] Confirma la operación presionando el botón \IUbutton{Si}.
    \UCpaso Muestra la interfaz de usuario menú para Jefe de Innovación Educativa (Pantalla de Abi)
\end{UCtrayectoriaA}


%------------------------ CU TRAYECTORIA ALTERNARIVA B -------------------------

\begin{UCtrayectoriaA}{B}{El actor presiona accidentalmente el botón Cancelar}
	\UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar}
    \UCpaso Muestra el mensaje \MSGref{MSG5}{¿Seguro que desea cancelar la consulta de Programas Académicos?}.
    \UCpaso[\UCactor] Presiona el botón \IUbutton{No}.
    \UCpaso Cierra el mensaje.
    \UCpaso Continúa en el paso 2 de la trayectoria principal del \UCref{SP4-CU5}.
\end{UCtrayectoriaA}

%------------------------ CU TRAYECTORIA ALTERNARIVA C -------------------------

\begin{UCtrayectoriaA}{C}{El usuario presiona el botón Registrar Programa Académico.}
	\UCpaso[\UCactor] Presiona el botón \IUbutton{Registrar Programa Académico}
    \UCpaso Muestra el mensaje \MSGref{MSG6}{¿Seguro que desea registrar un Programa Académico?}.
    \UCpaso[\UCactor] Cierra el mensaje presionando el botón \IUbutton{Aceptar}.
\UCpaso Muestra la interfaz de usuario \IUref{IU2.4.1-J}{Registrar Programas Académicos}
\end{UCtrayectoriaA}
