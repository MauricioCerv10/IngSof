\begin{table}[htbp!]
	\begin{requerimientos}
	
		\FRitem{RM1}{Registrar Mapa Curricular}{El Usuario Miembro de la Comisión puede registrar los datos del Mapa Curricular acorde al Modelo de Datos.}{A}{Origen}
	
		\FRitem{RM2}{Registrar Unidad de Aprendizaje}{El sistema debe permitir ingresar Unidades de Aprendizaje en el Mapa Curricular.}{A}{Origen}
		
        \FRitem{RM3}{Guardar Mapa Curricular}{El sistema debe permitir guardar el estado actual del Mapa Curricular .}{A}{Origen}
        
		\FRitem{RM4}{Finalizar Mapa Curricular}{El sistema debe permitir finalizar el registro del estado actual del Mapa  Curricular.}{A}{Origen}

		\FRitem{RM5}{Consultar Unidad de Aprendizaje }{El sistema debe permitir visualizar las Unidades de Aprendizaje del Mapa Curricular.}{M}{Origen}
		
		\FRitem{RM6}{Modificar Unidad de Aprendizaje }{El sistema debe permitir hacer modificaciones a cualquier Unidad de Aprendizaje previamente registrada.}{M}{Origen}
		
		 \FRitem{RM7}{Registrar dependencias de Unidad de Aprendizaje}{ El Sistema debe permitir el registro de dependencias de las Unidades de Aprendizaje.}{M}{Origen}

		 \FRitem{RM8}{Validar integridad de Mapa Curricular}{ El Sistema validara la Integridad de los datos del Mapa Curricular.}{M}{Origen}        
		
		\FRitem{RM9}{Validar Mapa Curricular}{ El Usuario Jefe  de Innovación Educativa  valida el estado del Mapa Curricular.}{M}{Origen}

        \FRitem{RM10}{Enviar Comentarios}{ El Sistema permite enviar comentarios  de la validación del Mapa Curricular por el  Usuario Jefe  de Innovación Educativa.}{M}{Origen}

        \FRitem{RM11}{Visualizar Comentarios}{ El Sistema permite visualizar comentarios  de la validación del Mapa Curricular enviados por el  Usuario Jefe de Innovación Educativa por el usuario Miembro de la Comisión.}{M}{Origen}
        
        
	\end{requerimientos}
    \caption{Requerimientos funcionales del sistema para el subproceso de elaboración del Mapa Curricular}
