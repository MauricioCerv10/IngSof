	\begin{requerimientos}
	
		\FRitem{RM1}{Registrar Mapa Curricular}{El Usuario Miembro de la Comisión puede registrar los datos del Mapa Curricular acorde al Modelo de Datos.}{A}{Origen}
	
		\FRitem{RM2}{Registrar Unidad de Aprendizaje}{El sistema debe permitir ingresar Unidades de Aprendizaje en el Mapa Curricular.}{A}{Origen}
		
        \FRitem{RM3}{Guardar Mapa Curricular}{El sistema debe permitir guardar el estado actual del Mapa Curricular .}{A}{Origen}
        
		\FRitem{RM4}{Finalizar Mapa Curricular}{El sistema debe permitir finalizar el registro del Mapa  Curricular.}{A}{Origen}

		\FRitem{RM5}{Consultar Unidad de Aprendizaje }{El sistema debe permitir visualizar las Unidades de Aprendizaje registradas.}{M}{Origen}
		
		\FRitem{RM6}{Modificar Unidad de Aprendizaje }{El sistema debe permitir hacer modificaciones a cualquiera de las Unidades de Aprendizaje registradas.}{M}{Origen}
		
		 \FRitem{RM7}{Gestionar dependencias de Unidad de Aprendizaje}{ El Sistema debe permitir la gestión de las dependencias entre las Unidades de Aprendizaje.}{M}{Origen}

		 \FRitem{RM8}{Validar congruencia en el Mapa Curricular}{ El Sistema validara la congruencia de los datos registrados.}{M}{Origen}        
		
		\FRitem{RM9}{Validar Mapa Curricular}{ El Usuario Jefe  de Innovación Educativa  valida el estado del Mapa Curricular.}{M}{Origen}

        \FRitem{RM10}{Enviar Comentarios}{ El Sistema debe permitirle al Usuario Jefe de Innovación Educativa enviar comentarios de corrección sobre el Mapa Curricular.}{M}{Origen}

        \FRitem{RM11}{Visualizar Comentarios}{ El Sistema debe permitirle al Usuario Miembro de la Comisión visualizar comentarios hechos al Mapa Curricular, tanto por el Departamento de Innovación Educativa como por la DES.}{M}{Origen}
        
        
	\end{requerimientos}
