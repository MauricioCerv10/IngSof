		\FRitem{RU}{Generar oficio de aprobación del programa académico}{El sistema debe permitir que la Jefa de la DES genere un oficio de que se aprobo el programa académico enviado por la Unidad Académica}{MA}{Origen}
		\FRitem{RU}{Elaborar un oficio de aprobación del plan de trabajo}{EL sistema debe permitir que la Jefa de la DES genere un oficio de que se aprobo el plan de trabajo}{MA}{Origen}
		\FRitem{RU}{Elaborar un oficio de aprobación del cronograma de actividades}{EL sistema debe permitir que la Jefa de la DES genere un oficio de que se aprobo el cronograma de actividades}{MA}{Origen}
		\FRitem{RU}{Enviar plantillas y documentos de apoyo para la elaboración del Plan de estudios}{El sistema debe permitir que la DES envie las plantillas y documentos de apoyo a la Unidad Académica}{}{}
		\FRitem{RU}{Enviar propuesta del Plan de Estudios}{El sistema debe permitir que la Unidad Académica envie la propuesta del Plan de Estudios a la DES}{MA}{Origen}
		\FRitem{RU}{Redactar acta de aprobación del Plan de Estudios}{El sistema debe permitir que la DES redacte una acta de aprobación del Plan de Estudios}{MA}{Origen}
		\FRitem{RU}{Generar dictamen técnico del Plan de Estudios}{El sistema debe permitir que la DES genere un dictamen técnico del plan de estudios, el cual será firmado por el director de la Unidad Académica}{MA}{Origen}
		\FRitem{RU}{Notificar aprobación de Plan de Estudios}{El sistema debe permitir notificar a la unidad académica la aprobación del plan de estudios por parte del Consejo General Consultivo}{MA}{Origen}
		\FRitem{RU}{Enviar plantillas y documentos de apoyo para la elaboración de las Unidades de Aprendizaje}{El sistema debe permitir que la DES envie las plantillas y documentos de apoyo a la Unidad Académica}{MA}{Origen}
		\FRitem{RU}{Enviar paquete de Unidades de Aprendizaje}{El sistema debe permitir que la Unidad Académica envie las Unidades de aprendizaje elaboradas a la DES}{MA}{Origen}
		\FRitem{RU}{Generar oficio por paquete de Unidades de Aprendizaje}{El sistema debe permitir generar un oficio por paquete de Unidades de Aprendizaje, indicando que se recibio dicho paquete}{MA}{Origen}

		\FRitem{RU}{Generar dictamen técnico de las Unidades de Aprendizaje}{El sistema debe permitir que la DES genere un dictamen técnico del paquete de Unidades de Aprendizaje aprobadas, el cual será firmado por el director de la Unidad Académica}{MA}{Origen}
		\FRitem{RU}{Notificar aprobación de las Unidades de Aprendizaje}{El sistema debe permitir notificar a la unidad académica la aprobación de la Unidad de Aprendizaje por parte del Consejo General Consultivo}{MA}{Origen}
